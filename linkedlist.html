Linked List

  • Linked list creates list of elements using next pointers.
  • It usually provides three methods getFirst, appendToTail and delete.
  • GetFirst returns the first element from the list, if there exists one.
  • AppendToTail appends an element to the tail of the list.
  • Delete deletes the element from the list, if it exits.
  • Some more variations of linked list are DoublyLinkedList and CircularLinkedList 

public class LinkedList {
    /** public class Node{
     * Node next;
     * Object data;	
     */
    private Node first;

    public LinkedList(Long data) {
        first = new Node(data);
    }

    public Node getFirst() {
        return first;
    }

    public void appendToTail(final Long data) {
        Node newNode = new Node(data);
        Node n = first;
        while (n.next != null) {
            n = n.next;
        }
        n.next = newNode;

    }

    public Long deleteNode(Long data) {
        Node n = first;
        if (n.data == data) {
            first = first.next;
            return data;
        }
        while (n.next != null) {
            if (n.next.data == data) {
                n.next = n.next.next;
                return data;
            }
            n = n.getNext();
        }
        return null;

    }
}
